84 static inline char *utoa(unsigned int num, char *str, int radix) {
85  char temp[17]; //an int can only be 16 bits long
86  //at radix 2 (binary) the string
87  //is at most 16 + 1 null long.
88  int temp_loc = 0;
89  int digit;
90  int str_loc = 0;
91 
92  //construct a backward string of the number.
93  do {
94  digit = (unsigned int)num % radix;
95  if (digit < 10)
96  temp[temp_loc++] = digit + '0';
97  else
98  temp[temp_loc++] = digit - 10 + 'A';
99  num = ((unsigned int)num) / radix;
100  } while ((unsigned int)num > 0);
101 
102  temp_loc--;
103 
104 
105  //now reverse the string.
106  while ( temp_loc >=0 ) {// while there are still chars
107  str[str_loc++] = temp[temp_loc--];
108  }
109  str[str_loc] = 0; // add null termination.
110 
111  return str;
112 }
