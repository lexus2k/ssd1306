60 void NanoCanvas::fillRect(uint8_t x1, uint8_t y1, uint8_t x2, uint8_t y2, uint8_t templ)
61 {
62  if ((x1 < x2) && (x1 >= m_w)) return;
63  if ((y1 < y2) && (y1 >= m_h)) return;
64  if (x1 > x2) x1 = 0;
65  if (y1 > y2) y1 = 0;
66  if (x2 >= m_w) x2 = m_w -1;
67  if (y2 >= m_h) y2 = m_h -1;
68  uint8_t bank1 = (y1 >> 3);
69  uint8_t bank2 = (y2 >> 3);
70  for (uint8_t bank = bank1; bank2<=(y2 >> 3); bank++)
71  {
72  uint8_t mask = 0xFF;
73  if (bank1 == bank2)
74  {
75  mask = (mask >> ((y1 & 7) + 7 - (y2 & 7))) << (7 - (y2 & 7));
76  }
77  else if (bank1 == bank)
78  {
79  mask = (mask >> (y1 & 7));
80  }
81  else if (bank2 == bank)
82  {
83  mask = (mask << (7 - (y2 & 7)));
84  }
85  for (uint8_t x=x1; x<x2; x++)
86  {
87  m_bytes[BADDR(bank) + x] &= ~mask;
88  m_bytes[BADDR(bank) + x] |= (templ & mask);
89  }
90  }
91 };

101 void NanoCanvas::char_f6x8(uint8_t x, uint8_t y, const char ch[])
102 {
103  uint8_t c, i, j;
104  if (y>=m_h) return;
105  j = 0;
106  while(ch[j] != '\0')
107  {
108  c = ch[j] - 32;
109  for(i=0;i<6;i++)
110  {
111  if (x>=m_w) return;
112  uint8_t d = pgm_read_byte(&ssd1306xled_font6x8[c*6+i]);
113  m_bytes[YADDR(y) + x] |= (d << (y & 0x7));
114  if (y+8 < m_h)
115  m_bytes[YADDR(y) + m_w + x] |= (d >> (8 - (y & 0x7)));
116  x++;
117  }
118  j++;
119  }
120 };
