184  inline SSD1306_RECT getUpdateRect() const
185  {
186  uint8_t left = min(x,lx);
187  uint8_t top = min(y,ly);
188  uint8_t right = max((uint8_t)(x + w - 1), (uint8_t)(lx + w - 1));
189  if (((uint8_t)(lx + w - 1) < w) && (right > 2*w))
190  {
191  right = (uint8_t)(lx + w - 1);
192  }
193  uint8_t bottom = max((uint8_t)(y + 7), (uint8_t)(ly + 7));
194  if (((uint8_t)(ly + 7) < 8) && (bottom > 16))
195  {
196  bottom = (uint8_t)(ly + 7);
197  }
198  if ( left > right ) left = 0;
199  if ( top > bottom ) top = 0;
200  return (SSD1306_RECT){ left, top, right, bottom };
