73 static void ssd1306_setBlock(lcduint_t x, lcduint_t y, lcduint_t w)
74 {
75  ssd1306_intf.start();
76  if (ssd1306_intf.spi)
77  ssd1306_spiDataMode(0);
78  else
79  ssd1306_intf.send(0x00);
80  ssd1306_intf.send(SSD1306_COLUMNADDR);
81  ssd1306_intf.send(x);
82  ssd1306_intf.send(w ? (x + w - 1) : (ssd1306_lcd.width - 1));
83  ssd1306_intf.send(SSD1306_PAGEADDR);
84  ssd1306_intf.send(y);
85  ssd1306_intf.send((ssd1306_lcd.height >> 3) - 1);
86  if (ssd1306_intf.spi)
87  {
88  ssd1306_spiDataMode(1);
89  }
90  else
91  {
92  ssd1306_intf.stop();
93  ssd1306_intf.start();
94  ssd1306_intf.send(0x40);
95  }
96 }

148 void ssd1306_128x64_spi_init(int8_t rstPin, int8_t cesPin, int8_t dcPin)
149 {
150  if (rstPin >=0)
151  {
152  pinMode(rstPin, OUTPUT);
153  digitalWrite(rstPin, HIGH);
154  /* Wait at least 1ms after VCC is up for LCD */
155  delay(1);
156  /* Perform reset operation of LCD display */
157  digitalWrite(rstPin, LOW);
158  delay(10);
159  digitalWrite(rstPin, HIGH);
160  }
161  ssd1306_spiInit(cesPin, dcPin);
162  ssd1306_128x64_init();
163 }
