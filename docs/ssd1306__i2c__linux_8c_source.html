60 static void ssd1306_i2cSendByte_Linux(uint8_t data)
61 {
62  s_buffer[s_dataSize] = data;
63  s_dataSize++;
64  if (s_dataSize == sizeof(s_buffer))
65  {
66  /* Send function puts all data to internal buffer. *
67  * Restart transmission if internal buffer is full. */
68  ssd1306_i2cStop_Linux();
69  ssd1306_dataStart();
70  }
71 }

90 void ssd1306_i2cInit_Linux(int8_t busId, uint8_t sa)
91 {
92  char filename[20];
93  if (busId < 0)
94  {
95  busId = 1;
96  }
97  snprintf(filename, 19, "/dev/i2c-%d", busId);
98  ssd1306_startTransmission = empty_function;
99  ssd1306_endTransmission = empty_function;
100  ssd1306_closeInterface = empty_function;
101  ssd1306_commandStart = empty_function;
102  ssd1306_dataStart = empty_function;
103  ssd1306_sendByte = empty_function_single_arg;
104  if ((s_fd = open(filename, O_RDWR)) < 0)
105  {
106  fprintf(stderr, "Failed to open the i2c bus\n");
107  return;
108  }
109  if (sa)
110  {
111  s_sa = sa;
112  }
113  if (ioctl(s_fd, I2C_SLAVE, s_sa) < 0)
114  {
115  fprintf(stderr, "Failed to acquire bus access and/or talk to slave.\n");
116  return;
117  }
118  ssd1306_startTransmission = ssd1306_i2cStart_Linux;
119  ssd1306_endTransmission = ssd1306_i2cStop_Linux;
120  ssd1306_sendByte = ssd1306_i2cSendByte_Linux;
121  ssd1306_closeInterface = ssd1306_i2cClose_Linux;
122  ssd1306_commandStart = ssd1306_i2cCommandStart;
123  ssd1306_dataStart = ssd1306_i2cDataStart;
124 }
