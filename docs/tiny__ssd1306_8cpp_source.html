32 size_t TinySSD1306::write(uint8_t ch)
33 {
34  const char str[2] = { static_cast<char>(ch), '\0' };
35  if (ch == '\r')
36  {
37  m_xpos = 0;
38  }
39  if ( (m_xpos > width() - FIXED_FONT_WIDTH) || (ch == '\n') )
40  {
41  m_xpos = 0;
42  m_ypos += FIXED_FONT_HEIGHT;
43  }
44  if ( m_ypos > (height() >> 3) - FIXED_FONT_HEIGHT )
45  {
46  m_ypos = 0;
47  }
48  ssd1306_printFixed( m_xpos, m_ypos, str, STYLE_NORMAL );
49  m_xpos += FIXED_FONT_WIDTH;
50 }
